          default: value

        break

        tmp = {
          type: String,
          default: value
        }
        break
      case 'boolean':
        tmp = {
          type: [Boolean, Number],
          default: value
        }
        break
      case 'object':
        if (value) {
          tmp = {
            type: Object,
            default () {
              return JSON.parse(JSON.stringify(value))
            }
          }
        } else {
          // Number String Function
          tmp = {
            type: [Object, Number, String, Function, Array],
            default () {
              return null
            }
          }

        console.debug('value: ' + value)

        break
    }
    props[key] = tmp
  }
  return props
