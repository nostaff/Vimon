<script type="text/javascript">
  export default{
    name: 'comp',
    data () {
      return {
        contentEle: null,
        fontFamily: '',
        background: '',
        colors: {
          'white': {
            'bg': 'rgb(255, 255, 255)',
            'fg': 'rgb(0, 0, 0)'
          },
          'tan': {
            'bg': 'rgb(249, 241, 228)',
            'fg': 'rgb(0, 0, 0)'
          },
          'grey': {
            'bg': 'rgb(76, 75, 80)',
            'fg': 'rgb(255, 255, 255)'
          },
          'black': {
            'bg': 'rgb(0, 0, 0)',
            'fg': 'rgb(255, 255, 255)'
          }
        }
      }
    },
    methods: {
      changeBackground (color) {
        this.background = color
        this.contentEle.style.backgroundColor = this.colors[color].bg
        this.contentEle.style.color = this.colors[color].fg
      },
      changeFontFamily () {
        if (this.fontFamily) this.contentEle.style.fontFamily = this.fontFamily
      },
      changeFontSize (direction) {
        this.contentEle.style.fontSize = direction
      },
      getColorName (background) {
        let colorName = 'white'
        if (!background) return 'white'
        for (var key in this.colors) {
          if (this.colors[key].bg === background) {
            colorName = key
          }
        }
        return colorName
      },
      setFontFamily () {
        if (this.contentEle.style.fontFamily) {
          this.fontFamily = this.contentEle.style.fontFamily.replace(/'/g, '')
        }
      }
    },
    created () {
      this.contentEle = this.$options.$data.contentEle
      this.background = this.getColorName(this.contentEle.style.backgroundColor)
      this.setFontFamily()
    }
  }
</script>
